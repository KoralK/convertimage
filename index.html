WebP to JPEG Converter

from pyodide import create_proxy from PIL import Image from js import document, fetch from io import BytesIO import base64 import asyncio async def install_pillow(): import micropip await micropip.install('pillow') async def convert_to_jpeg(file): webp_image = Image.open(BytesIO(await file.arrayBuffer().to_py())) jpeg_image_bytes = BytesIO() webp_image.save(jpeg_image_bytes, format='JPEG') jpeg_base64 = base64.b64encode(jpeg_image_bytes.getvalue()).decode('utf-8') document.getElementById('output-image').src = f"data:image/jpeg;base64,{jpeg_base64}" document.getElementById('output-image').style.display = 'block' async def handle_file_upload(event): await install_pillow() # Ensure Pillow is installed file_input = document.getElementById('upload-file') if file_input.files.length > 0: file = file_input.files[0] await convert_to_jpeg(file) # Create a Python proxy for the JavaScript function and assign it to the button's onclick event convertToJPEG = create_proxy(handle_file_upload) document.querySelector('button').onclick = convertToJPEG
