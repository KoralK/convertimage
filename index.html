WebP to JPEG Converter

from asyncio import ensure_future from PIL import Image from js import document, fetch, window from io import BytesIO import base64 import micropip async def install_pillow(): await micropip.install('pillow') async def convert_to_jpeg(file): await install_pillow() # Ensure Pillow is installed inside the coroutine webp_image = Image.open(BytesIO(await file.arrayBuffer().to_py())) jpeg_image_bytes = BytesIO() webp_image.save(jpeg_image_bytes, format='JPEG') jpeg_base64 = base64.b64encode(jpeg_image_bytes.getvalue()).decode('utf-8') document.getElementById('output-image').src = f"data:image/jpeg;base64,{jpeg_base64}" document.getElementById('output-image').style.display = 'block' def on_upload_click(event): file_input = document.getElementById('upload-file') if file_input.files.length > 0: file = file_input.files[0] ensure_future(convert_to_jpeg(file)) document.getElementById('upload-button').addEventListener('click', on_upload_click)
