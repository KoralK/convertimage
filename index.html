WebP to JPEG Converter

import asyncio import micropip await micropip.install('pillow') from PIL import Image from js import document from io import BytesIO import base64 def convert_to_jpeg(webp_data): webp_image = Image.open(BytesIO(webp_data)) jpeg_image_bytes = BytesIO() webp_image.save(jpeg_image_bytes, format='JPEG') jpeg_base64 = base64.b64encode(jpeg_image_bytes.getvalue()).decode('utf-8') return jpeg_base64 def file_to_bytes(file): return file.arrayBuffer().to_py() async def convertToJPEG(): file_input = document.getElementById('upload-file') if file_input.files.length == 0: return file = file_input.files[0] file_bytes = await file_to_bytes(file) jpeg_base64 = convert_to_jpeg(file_bytes) output_img = document.getElementById('output-image') output_img.src = f"data:image/jpeg;base64,{jpeg_base64}" output_img.style.display = 'block' asyncio.ensure_future(convertToJPEG())
